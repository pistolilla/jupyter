DROP TABLE IF EXISTS utterance;
CREATE TABLE IF NOT EXISTS utterance (cId INTEGER NOT NULL, uId INTEGER NOT NULL, Utterance TEXT NOT NULL);
DROP TABLE IF EXISTS utterance_act;
CREATE TABLE IF NOT EXISTS utterance_act (cId INTEGER NOT NULL, uId INTEGER NOT NULL, actId INTEGER NOT NULL);
DROP TABLE IF EXISTS utterance_emotion;
CREATE TABLE IF NOT EXISTS utterance_emotion (cId INTEGER NOT NULL, uId INTEGER NOT NULL, emotionId INTEGER NOT NULL);
DROP TABLE IF EXISTS conversation_topic;
CREATE TABLE IF NOT EXISTS conversation_topic (cId INTEGER NOT NULL, uId INTEGER NOT NULL, topicId INTEGER NOT NULL, UNIQUE(cId, topicId) ON CONFLICT IGNORE);
